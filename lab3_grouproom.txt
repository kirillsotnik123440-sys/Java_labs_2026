import java.util.Arrays;

public class GroupRoom {
    private int groupId;
    private Room[] rooms;
    public GroupRoom() {
        this.groupId = 0;
        this.rooms = new Room[0];
    }
    public GroupRoom(int groupId, Room[] rooms) {
        this.groupId = groupId;
        this.rooms = rooms;
    }
    public Room[] getRooms() {
        return rooms;
    }
    public void setRooms(Room[] rooms) {
        this.rooms = rooms;
    }
    public Room getRoom(int index) {
        return rooms[index];
    }
    public void setRoom(int index, Room room) {
        rooms[index] = room;
    }
    public void addRoom(Room room) {
        rooms = Arrays.copyOf(rooms, rooms.length + 1);
        rooms[rooms.length - 1] = room;
    }
    public void removeRoomById(int id) {
        Room[] temp = new Room[rooms.length - 1];
        int index = 0;
        for (Room room : rooms) {
            if (room.getId() != id) {
                if (index < temp.length) {
                    temp[index++] = room;
                }
            }
        }
        rooms = temp;
    }
    public void sortByPrice() {
        for (int i = 0; i < rooms.length - 1; i++) {
            for (int j = i + 1; j < rooms.length; j++) {
                if (rooms[i].getPrice() > rooms[j].getPrice()) {
                    Room temp = rooms[i];
                    rooms[i] = rooms[j];
                    rooms[j] = temp;
                }
            }
        }
    }
    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder("GroupRoom ID=" + groupId + "\n");
        for (Room room : rooms) {
            sb.append(room).append("\n");
        }
        return sb.toString();
    }
}
