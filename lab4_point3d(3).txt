public class Point3D extends Point {
    private int z;  // Координата Z

    // Конструктор без параметров, точка в (0, 0, 0)
    public Point3D() {
        super();  // Вызов конструктора родительского класса (Point)
        this.z = 0;
    }

    // Конструктор с параметрами для координат (x, y, z)
    public Point3D(int x, int y, int z) {
        super(x, y);  // Вызов конструктора родительского класса (Point) для x и y
        this.z = z;
    }

    // Установка новых координат (x, y, z)
    public void setLocation(int x, int y, int z) {
        super.setLocation(x, y);  // Вызов метода родительского класса для x и y
        this.z = z;
    }

    // Установка новых координат (x, y), z устанавливается в 0
    @Override
    public void setLocation(int x, int y) {
        super.setLocation(x, y);  // Вызов метода родительского класса для x и y
        this.z = 0;  // Координата Z устанавливается в 0
    }

    // Получение координаты Z
    public int getZ() {
        return z;
    }

    // Преобразование точки в строку "(x, y, z)"
    @Override
    public String toString() {
        return "(" + getX() + "," + getY() + "," + z + ")";
    }

    // Расстояние от точки до начала координат (0, 0, 0) с учетом Z
    @Override
    public int distanceFromOrigin() {
        return (int) Math.sqrt(getX() * getX() + getY() * getY() + z * z);
    }

    // Проверка, находится ли точка внутри трехмерного прямоугольного параллелепипеда
    public boolean isInBox(Point3D corner1, Point3D corner2) {
        return getX() >= Math.min(corner1.getX(), corner2.getX()) && getX() <= Math.max(corner1.getX(), corner2.getX()) &&
               getY() >= Math.min(corner1.getY(), corner2.getY()) && getY() <= Math.max(corner1.getY(), corner2.getY()) &&
               z >= Math.min(corner1.getZ(), corner2.getZ()) && z <= Math.max(corner1.getZ(), corner2.getZ());
    }

    // Проверка, находится ли точка внутри сферы
    public boolean isInSphere(Point3D center, int radius) {
        int distanceSquared = (getX() - center.getX()) * (getX() - center.getX()) +
                              (getY() - center.getY()) * (getY() - center.getY()) +
                              (z - center.getZ()) * (z - center.getZ());
        return distanceSquared <= radius * radius;
    }
}
