import java.util.Random;
import java.util.Scanner;

public class Main {
    public static double[] CreateReciprocalFraction(int[] array) {
        double[] result = new double[array.length];
        for (int i = 0; i < array.length; i++) {
            try {
                result[i] = 1.0 / array[i];
            } catch (ArithmeticException e) {
                System.out.println("Ошибка: деление на ноль в элементе с индексом " + i);
                result[i] = 0;
            }
        }
        return result;
    }
    public static void main(String[] args) {
        Random random = new Random();
        Scanner sc = new Scanner(System.in);
        System.out.print("Введите размер массива n: ");
        int n = sc.nextInt();
        int[] array = new int[n];
        for (int i = 0; i < array.length; i++) {
            array[i] = random.nextInt(99) - 99;
        }
        System.out.println("Исходный массив:");
        for (int x : array) {
            System.out.print(x + " ");
        }
        System.out.println();
        double[] reciprocalArray = CreateReciprocalFraction(array);
        System.out.println("Массив обратных значений:");
        for (double x : reciprocalArray) {
            System.out.print(x + " ");
        }
        System.out.println();
        int[] allowedValues = { -5, -1, 1, 2, 5 };
        int count = 0;
        for (int value : array) {
            for (int allowed : allowedValues) {
                if (value == allowed) {
                    count++;
                    break;
                }
            }
        }
        System.out.println("Количество элементов с разрешёнными значениями: " + count);
    }
}
