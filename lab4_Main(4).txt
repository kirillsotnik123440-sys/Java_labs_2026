public class Main {
    public static void main(String[] args) {
        // Создаем точку в 2D
        Point point2D = new Point(3, 4);
        System.out.println("Точка 2D: " + point2D);
        System.out.println("Длина вектора от начала координат: " + point2D.vectorLengthFromOrigin());

        // Создаем точку в 3D
        Point3D point3D = new Point3D(3, 4, 5);
        System.out.println("Точка 3D: " + point3D);
        System.out.println("Длина вектора от начала координат: " + point3D.vectorLengthFromOrigin());
    }
}
